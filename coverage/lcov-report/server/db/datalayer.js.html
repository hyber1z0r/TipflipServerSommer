    var deferred = Q.defer();
 
    var cat = new Category({
        name: name ? name.capitalize(true).trim() : name,
        imagePath: imagePath,
        contentType: contentType
    });
 
    cat.save(function (err, category) {
        if (err) {
            deferred.reject(err);
